# 非对称加密

你想让通信双方安全交换密钥，但不知道怎么做。

问题出在哪？**没有正确使用非对称加密**。

非对称加密是现代密码学的基石。

│  │                                                         │   │

## RSA 实现

```java
// RSA 密钥对生成
@Service
public class RsaKeyService {
    private static final int KEY_SIZE = 2048;

    // 生成密钥对
    public KeyPair generateKeyPair() throws NoSuchAlgorithmException {
        KeyPairGenerator generator = KeyPairGenerator.getInstance("RSA");
        generator.initialize(KEY_SIZE);
        return generator.generateKeyPair();
    }

    // RSA 加密
    public String encrypt(String plaintext, PublicKey publicKey)
            throws Exception {
        Cipher cipher = Cipher.getInstance("RSA/ECB/OAEPWithSHA-256AndMGF1Padding");
        cipher.init(Cipher.ENCRYPT_MODE, publicKey);

        // RSA 一次只能加密小于密钥长度的数据
        byte[] plaintextBytes = plaintext.getBytes(StandardCharsets.UTF_8);
        int inputLen = plaintextBytes.length;
        int maxLen = KEY_SIZE / 8 - 42;  // OAEPWithSHA-256 填充后可用长度

        ByteArrayOutputStream output = new ByteArrayOutputStream();
        for (int i = 0; i < inputLen; i += maxLen) {
            byte[] chunk = Arrays.copyOfRange(
                plaintextBytes, i, Math.min(i + maxLen, inputLen));
            byte[] encrypted = cipher.doFinal(chunk);
            output.writeBytes(encrypted);
        }

        return Base64.getEncoder().encodeToString(output.toByteArray());
    }

    // RSA 解密
    public String decrypt(String ciphertext, PrivateKey privateKey)
            throws Exception {
        Cipher cipher = Cipher.getInstance("RSA/ECB/OAEPWithSHA-256AndMGF1Padding");
        cipher.init(Cipher.DECRYPT_MODE, privateKey);

        byte[] ciphertextBytes = Base64.getDecoder().decode(ciphertext);
        int maxLen = KEY_SIZE / 8;

        ByteArrayOutputStream output = new ByteArrayOutputStream();
        for (int i = 0; i < ciphertextBytes.length; i += maxLen) {
            byte[] chunk = Arrays.copyOfRange(
                ciphertextBytes, i, Math.min(i + maxLen, ciphertextBytes.length));
            byte[] decrypted = cipher.doFinal(chunk);
            output.writeBytes(decrypted);
        }

        return new String(output.toByteArray(), StandardCharsets.UTF_8);
    }

    // 导出公钥（Base64 PEM 格式）
    public String exportPublicKey(PublicKey publicKey) {
        return Base64.getEncoder().encodeToString(publicKey.getEncoded());
    }

    // 导出私钥（Base64 PEM 格式）
    public String exportPrivateKey(PrivateKey privateKey) {
        return Base64.getEncoder().encodeToString(privateKey.getEncoded());
    }

    // 导入公钥
    public PublicKey importPublicKey(String base64Key) throws Exception {
        byte[] keyBytes = Base64.getDecoder().decode(base64Key);
        X509EncodedKeySpec keySpec = new X509EncodedKeySpec(keyBytes);
        KeyFactory factory = KeyFactory.getInstance("RSA");
        return factory.generatePublic(keySpec);
    }

    // 导入私钥
    public PrivateKey importPrivateKey(String base64Key) throws Exception {
        byte[] keyBytes = Base64.getDecoder().decode(base64Key);
        PKCS8EncodedKeySpec keySpec = new PKCS8EncodedKeySpec(keyBytes);
        KeyFactory factory = KeyFactory.getInstance("RSA");
        return factory.generatePrivate(keySpec);
    }
}
```

## 签名实现

```java
// 数字签名
@Service
public class SignatureService {
    private static final String SIGNATURE_ALGORITHM = "SHA256withRSA";

    // 签名
    public String sign(String message, PrivateKey privateKey)
            throws Exception {
        Signature signature = Signature.getInstance(SIGNATURE_ALGORITHM);
        signature.initSign(privateKey);
        signature.update(message.getBytes(StandardCharsets.UTF_8));
        byte[] signatureBytes = signature.sign();
        return Base64.getEncoder().encodeToString(signatureBytes);
    }

    // 验签
    public boolean verify(String message, String signatureStr,
                         PublicKey publicKey) throws Exception {
        Signature signature = Signature.getInstance(SIGNATURE_ALGORITHM);
        signature.initVerify(publicKey);
        signature.update(message.getBytes(StandardCharsets.UTF_8));
        byte[] signatureBytes = Base64.getDecoder().decode(signatureStr);
        return signature.verify(signatureBytes);
    }

    // 签名 + 验签示例
    public void demo() throws Exception {
        // 生成密钥对
        KeyPair keyPair = rsaKeyService.generateKeyPair();
        PublicKey publicKey = keyPair.getPublic();
        PrivateKey privateKey = keyPair.getPrivate();

        String message = "Hello, Bob!";

        // 签名
        String signature = sign(message, privateKey);
        System.out.println("Signature: " + signature);

        // 验签
        boolean valid = verify(message, signature, publicKey);
        System.out.println("Verification: " + valid);  // true
    }
}
```

### 混合加密实现

```java
// 混合加密服务
@Service
public class HybridEncryptionService {
    @Autowired
    private AesGcmEncryptionService aesService;

    @Autowired
    private RsaKeyService rsaService;

    // 发送方加密
    public EncryptedPackage encrypt(String plaintext, PublicKey recipientPublicKey)
            throws Exception {
        // 1. 生成随机对称密钥
        SecretKey aesKey = generateAesKey();

        // 2. 用对称密钥加密数据
        String encryptedData = aesService.encrypt(plaintext, aesKey);

        // 3. 用 RSA 公钥加密对称密钥
        String encryptedKey = rsaService.encrypt(
            Base64.getEncoder().encodeToString(aesKey.getEncoded()),
            recipientPublicKey);

        return new EncryptedPackage(encryptedData, encryptedKey);
    }

    // 接收方解密
    public String decrypt(EncryptedPackage pkg, PrivateKey privateKey)
            throws Exception {
        // 1. 用 RSA 私钥解密对称密钥
        String aesKeyBase64 = rsaService.decrypt(pkg.getEncryptedKey(), privateKey);
        byte[] aesKeyBytes = Base64.getDecoder().decode(aesKeyBase64);
        SecretKey aesKey = new SecretKeySpec(aesKeyBytes, "AES");

        // 2. 用对称密钥解密数据
        return aesService.decrypt(pkg.getEncryptedData(), aesKey);
    }

    private SecretKey generateAesKey() throws NoSuchAlgorithmException {
        KeyGenerator keyGen = KeyGenerator.getInstance("AES");
        keyGen.init(256);
        return keyGen.generateKey();
    }

    @Data
    @AllArgsConstructor
    public static class EncryptedPackage {
        private String encryptedData;
        private String encryptedKey;
    }
}
```

## 总结

非对称加密是现代密码学的基石。

**核心概念**：

- 公钥加密，私钥解密
- 私钥签名，公钥验签

**RSA 算法**：

- 基于大数分解的困难性
- 密钥长度：2048 位以上

**应用场景**：

- 密钥交换
- 数字签名
- 混合加密

**最佳实践**：

- 大数据用对称加密
- 密钥交换用非对称加密
- 私钥必须安全存储

**核心认知**：对称加密和非对称加密各有优势，实际系统中通常混合使用，取长补短。
