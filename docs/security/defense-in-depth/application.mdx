# 应用层防护

你的应用没有过滤用户输入，攻击者通过 SQL 注入拖走了整个数据库。

问题出在哪？**没有应用层防护**。

应用层是安全防护的核心层。

## SQL 注入防护

```java
// 错误示范：字符串拼接 SQL
@Service
public class BadSqlInjectionExample {
    @Autowired
    private JdbcTemplate jdbcTemplate;

    // 极度危险！
    public User findUser(String username) {
        // 直接拼接用户输入
        String sql = "SELECT * FROM users WHERE username = '" + username + "'";
        return jdbcTemplate.queryForObject(sql, userMapper);
    }
    // 攻击：username = "' OR '1'='1' --"
    // SQL = SELECT * FROM users WHERE username = '' OR '1'='1' --'
    // 结果：返回所有用户
}

// 正确做法：参数化查询
@Service
public class SqlInjectionPrevention {
    @Autowired
    private JdbcTemplate jdbcTemplate;

    public User findUser(String username) {
        // 参数化查询
        String sql = "SELECT * FROM users WHERE username = ?";
        return jdbcTemplate.queryForObject(sql,
            new Object[]{username},
            userMapper
        );
    }

    // MyBatis 使用 #{} 而非 ${}
    // Mapper:
    // SELECT * FROM users WHERE username = #{username}
    // 而非:
    // SELECT * FROM users WHERE username = ${username}
}

// JPA 使用参数化查询
@Repository
public interface UserRepository extends JpaRepository<User, Long> {
    // 自动使用参数化查询
    Optional<User> findByUsername(String username);

    @Query("SELECT u FROM User u WHERE u.email = ?1")
    Optional<User> findByEmail(String email);
}
```

## XSS 防护

```java
// XSS 防护：输入验证 + 输出编码
@Service
public class XssPrevention {

    // Spring Security Header 配置
    @Configuration
    @EnableWebSecurity
    public class SecurityHeadersConfig {
        @Bean
        public SecurityFilterChain filterChain(HttpSecurity http) throws Exception {
            return http
                // Content Security Policy
                .headers(headers -> headers
                    .contentSecurityPolicy(csp -> csp
                        .policyDirectives(
                            "default-src 'self'; " +
                            "script-src 'self' 'nonce-{random}'; " +
                            "object-src 'none'; " +
                            "base-uri 'self';"
                        )
                    )
                    // X-XSS-Protection
                    .xssProtection(xss -> xss
                        .headerValue(XXssProtectionRequestMatcher.ENABLE)
                    )
                    // X-Content-Type-Options
                    .contentTypeOptions(contentType -> {})
                    // X-Frame-Options
                    .frameOptions(frame -> frame
                        .deny()
                    )
                )
                .build();
        }
    }

    // HTML 转义工具
    public String escapeHtml(String input) {
        if (input == null) {
            return null;
        }
        return HtmlUtils.htmlEscape(input);
    }

    // 富文本处理（允许部分 HTML）
    public String sanitizeHtml(String input) {
        // 使用 OWASP Java HTML Sanitizer
        PolicyFactory policy = new HtmlPolicyBuilder()
            .allowElements("p", "br", "b", "i", "u", "em", "strong")
            .allowAttributes("href").onElements("a")
            .requireRelNoopener()
            .toFactory();

        return policy.sanitize(input);
    }
}

// 前端 XSS 防护
// React（默认转义）
function UserDisplay({ user }) {
    // React 默认会转义，安全的
    return <div>{user.bio}</div>;
}

// Vue（需要手动处理）
function UserDisplay({ user }) {
    // 使用 v-text 或 {{}} 自动转义
    return <div>{{ user.bio }}</div>;

    // 避免使用 v-html，除非内容可信
    // return <div v-html="user.bio"></div>;
}
```

## CSRF 防护

```java
// CSRF 防护配置
@Configuration
@EnableWebSecurity
public class CsrfConfig {

    @Bean
    public SecurityFilterChain filterChain(HttpSecurity http) throws Exception {
        return http
            // 启用 CSRF 防护
            .csrf(csrf -> csrf
                // SPA 使用 Cookie Token
                .csrfTokenRepository(CookieCsrfTokenRepository.withHttpOnlyFalse())
                // 排除公共 API
                .ignoringAntMatchers("/api/public/**")
            )
            // SameSite Cookie
            .sessionManagement(session -> session
                .cookieSecurityPolicy(cookie -> cookie
                    .sameSite("Strict")
                )
            )
            .build();
    }
}

// Angular CSRF
// app.module.ts
@NgModule({
  imports: [
    HttpClientModule,
    HttpClientXsrfModule.withOptions({
      cookieName: 'XSRF-TOKEN',
      headerName: 'X-XSRF-TOKEN'
    })
  ]
})
```

## 越权访问防护

```java
// 越权访问防护
@Service
public class AccessControlService {

    // 水平越权防护：只能访问自己的数据
    public Order getOrder(Long userId, Long orderId) {
        Order order = orderRepository.findById(orderId);

        if (order == null) {
            throw new ResourceNotFoundException("订单不存在");
        }

        // 核心检查：订单属于当前用户
        if (!order.getUserId().equals(userId)) {
            // 记录安全事件
            securityEventService.record(
                SecurityEventType.HORIZONTAL_PRIVILEGE_ESCALATION,
                userId,
                "尝试访问他人订单: " + orderId
            );
            throw new ForbiddenException("无权访问此订单");
        }

        return order;
    }

    // 垂直越权防护：检查用户角色
    @PreAuthorize("hasRole('ADMIN')")
    public void deleteUser(Long userId) {
        userRepository.deleteById(userId);
    }

    // 更细粒度的权限检查
    @PreAuthorize("@accessControl.canDeleteUser(#currentUserId, #targetUserId)")
    public void deleteUser(Long currentUserId, Long targetUserId) {
        // 检查当前用户是否可以删除目标用户
        // 例如：管理员可以删除普通用户，但不能删除其他管理员
        userRepository.deleteById(targetUserId);
    }
}

// 全局异常处理
@RestControllerAdvice
public class AccessDeniedExceptionHandler {

    @ExceptionHandler(ForbiddenException.class)
    public ResponseEntity<ErrorResponse> handleForbidden(ForbiddenException e) {
        // 不泄露详细信息
        return ResponseEntity.status(403)
            .body(new ErrorResponse("FORBIDDEN", "Access denied"));
    }
}
```

## API 安全

```java
// API 安全配置
@Configuration
public class ApiSecurityConfig {

    // API 限流
    @Bean
    public FilterRegistrationBean<RateLimitFilter> rateLimitFilter() {
        FilterRegistrationBean<RateLimitFilter> bean = new FilterRegistrationBean<>();
        bean.setFilter(new RateLimitFilter());
        bean.addUrlPatterns("/api/*");
        return bean;
    }

    // API 版本控制
    @RestController
    @RequestMapping("/api/v{version}")
    public class ApiController {

        @GetMapping("/users")
        public ResponseEntity<?> getUsers(
                @PathVariable String version) {

            // 验证 API 版本
            if (!supportedVersions.contains(version)) {
                return ResponseEntity.status(400)
                    .body("Unsupported API version");
            }

            return ResponseEntity.ok(userService.findAll());
        }
    }

    // 请求体大小限制
    @Configuration
    public class RequestSizeConfig {
        @Bean
        public FilterRegistrationBean<RequestSizeFilter> requestSizeFilter() {
            FilterRegistrationBean<RequestSizeFilter> bean = new FilterRegistrationBean<>();
            bean.setFilter(new RequestSizeFilter(10 * 1024 * 1024)); // 10MB
            return bean;
        }
    }
}

// Rate Limit Filter 实现
public class RateLimitFilter implements Filter {
    private final Map<String, AtomicInteger> counters = new ConcurrentHashMap<>();
    private final int maxRequests;
    private final Duration window;

    public RateLimitFilter() {
        this.maxRequests = 100;
        this.window = Duration.ofMinutes(1);
    }

    @Override
    public void doFilter(ServletRequest request, ServletResponse response,
                        FilterChain chain) throws IOException, ServletException {
        HttpServletRequest httpRequest = (HttpServletRequest) request;
        String key = getClientKey(httpRequest);

        AtomicInteger counter = counters.computeIfAbsent(key,
            k -> new AtomicInteger(0));

        if (counter.incrementAndGet() > maxRequests) {
            HttpServletResponse httpResponse = (HttpServletResponse) response;
            httpResponse.setStatus(429);
            httpResponse.getWriter().write("Too many requests");
            return;
        }

        chain.doFilter(request, response);
    }
}
```

## 总结

应用层是安全防护的核心层。

**核心防护**：

- SQL 注入：参数化查询
- XSS：输入验证 + 输出编码
- CSRF：Token 验证 + SameSite Cookie
- 越权访问：显式权限检查
- API 安全：限流 + 版本控制

**安全编码原则**：

- 输入验证：所有用户输入都不可信
- 输出编码：根据上下文选择编码方式
- 参数化查询：永远不用字符串拼接 SQL
- 最小权限：只授予必要的权限

**核心认知**：应用层是最容易被攻击的层，也是防护效果最明显的层。再好的网络层防火墙，也挡不住应用层的注入漏洞。
