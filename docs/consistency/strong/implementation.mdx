## 单主复制实现

### MySQL 主从复制

```sql
-- MySQL 主从复制配置

-- 主库配置
[mysqld]
server-id = 1
log-bin = mysql-bin
binlog_format = ROW
sync_binlog = 1          -- 同步 binlog（强一致性）
gtid_mode = ON           -- GTID 模式
enforce_gtid_consistency = ON

-- 从库配置
[mysqld]
server-id = 2
relay-log = relay-bin
read_only = ON
gtid_mode = ON
```

```java
// 强一致性读：强制读主库
@Service
public class StrongReadService {
    @Autowired
    private DataSource masterDataSource;

    @Autowired
    private DataSource slaveDataSource;

    // 强一致性读：读主库
    public Object strongRead(String sql) {
        try (Connection conn = masterDataSource.getConnection();
             PreparedStatement stmt = conn.prepareStatement(sql)) {
            return stmt.executeQuery();
        } catch (SQLException e) {
            throw new RuntimeException(e);
        }
    }

    // 普通读：可以读从库
    public Object normalRead(String sql) {
        try (Connection conn = slaveDataSource.getConnection();
             PreparedStatement stmt = conn.prepareStatement(sql)) {
            return stmt.executeQuery();
        } catch (SQLException e) {
            throw new RuntimeException(e);
        }
    }
}
```

### Raft 算法核心


### Raft 实现示例

```java
// Raft 核心实现
public class RaftNode {
    private volatile State state = State.FOLLOWER;
    private volatile long currentTerm = 0;
    private volatile String votedFor = null;
    private volatile List<LogEntry> log = new ArrayList<>();
    private volatile int commitIndex = 0;
    private volatile int lastApplied = 0;

    // 投票
    public VoteResponse handleVoteRequest(VoteRequest request) {
        if (request.getTerm() > currentTerm) {
            currentTerm = request.getTerm();
            state = State.FOLLOWER;
            votedFor = null;
        }

        // 判断是否投票
        boolean granted = false;
        if (request.getTerm() == currentTerm &&
            (votedFor == null || votedFor.equals(request.getCandidateId())) &&
            isLogUpToDate(request.getLastLogIndex(), request.getLastLogTerm())) {
            granted = true;
            votedFor = request.getCandidateId();
        }

        return new VoteResponse(currentTerm, granted);
    }

    // 日志复制
    public AppendEntriesResponse handleAppendEntries(AppendEntriesRequest request) {
        if (request.getTerm() > currentTerm) {
            currentTerm = request.getTerm();
            state = State.FOLLOWER;
        }

        boolean success = false;
        if (request.getTerm() == currentTerm) {
            state = State.LEADER;
            success = appendEntries(request);
        }

        return new AppendEntriesResponse(currentTerm, success);
    }

    // 复制日志到从节点
    public void replicateLog() {
        for (String follower : config.getOtherNodes()) {
            // 获取需要复制的日志
            List<LogEntry> entries = getEntriesSince(
                nextIndex.get(follower));

            AppendEntriesRequest request = AppendEntriesRequest.builder()
                .term(currentTerm)
                .leaderId(nodeId)
                .prevLogIndex(prevLogIndex)
                .prevLogTerm(prevLogTerm)
                .entries(entries)
                .leaderCommit(commitIndex)
                .build();

            AppendEntriesResponse response = sendAppendEntries(follower, request);

            if (response.isSuccess()) {
                nextIndex.put(follower, prevLogIndex + entries.size());
                matchIndex.put(follower, prevLogIndex + entries.size());
            }
        }

        // 提交日志（需要多数派确认）
        for (int i = commitIndex + 1; i <= log.size(); i++) {
            if (log.get(i).getTerm() == currentTerm) {
                int count = 1;  // 自身
                for (int match : matchIndex.values()) {
                    if (match >= i) count++;
                }
                if (count > config.getQuorum()) {
                    commitIndex = i;
                }
            }
        }
    }
}
```

### ZooKeeper 分布式锁

```java
// ZooKeeper 分布式锁实现
@Service
public class ZookeeperLock {
    @Autowired
    private ZooKeeper zooKeeper;

    private final String lockPath;

    public boolean acquire() throws InterruptedException, KeeperException {
        // 创建临时顺序节点
        String nodePath = zooKeeper.create(
            lockPath + "/lock-",
            new byte[0],
            ZooDefs.Ids.OPEN_ACL_UNSAFE,
            CreateMode.EPHEMERAL_SEQUENTIAL);

        // 获取所有子节点并排序
        List<String> children = zooKeeper.getChildren(lockPath, false);
        Collections.sort(children);

        String nodeName = nodePath.substring(nodePath.lastIndexOf('/') + 1);

        // 如果是最小的节点，获取锁
        if (children.get(0).equals(nodeName)) {
            return true;
        }

        // 否则监听前一个节点
        String prevNode = lockPath + "/" + children.get(
            children.indexOf(nodeName) - 1);

        CountDownLatch latch = new CountDownLatch(1);
        Stat stat = zooKeeper.exists(prevNode, event -> {
            if (event.getType() == Watcher.Event.EventType.NodeDeleted) {
                latch.countDown();
            }
        });

        // 等待前一个节点删除
        latch.await();
        return true;
    }

    public void release() throws InterruptedException, KeeperException {
        // 删除临时节点
        zooKeeper.delete(lockPath, -1);
    }
}
```

## 多副本同步复制

### Quorum 机制

```
┌─────────────────────────────────────────────────────────────────┐
│                    Quorum 机制                                    │
├─────────────────────────────────────────────────────────────────┤
│                                                                  │
│  写入需要 W 个节点确认                                          │
│  读取需要 R 个节点                      │
│  只要 W + R > N，保证读取到最新数据                            │
│                                                                  │
│  ┌─────────────────────────────────────────────────────────┐   │
│  │                                                         │   │
│  │  示例：N = 3                                           │   │
│  │                                                         │   │
│  │  方案 1：W = 2, R = 2                                 │   │
│  │  写入需要 2 个节点确认，读取需要 2 个节点             │   │
│  │  W + R = 4 > 3，保证强一致性                         │   │
│  │                                                         │   │
│  │  方案 2：W = 1, R = 3                                 │   │
│  │  写入需要 1 个节点确认，读取需要 3 个节点             │   │
│  │  适合读多写少                                         │   │
│  │                                                         │   │
│  │  方案 3：W = 3, R = 1                                 │   │
│  │  写入需要 3 个节点确认，读取需要 1 个节点             │   │
│  │  适合写多读少                                         │   │
│  │                                                         │   │
│  └─────────────────────────────────────────────────────────┘   │
│                                                                  │
└─────────────────────────────────────────────────────────────────┘
```

```java
// Quorum 读写实现
@Service
public class QuorumKVStore {
    private final List<Node> nodes;
    private final int N = 3;       // 总副本数
    private final int W = 2;        // 写入确认数
    private final int R = 2;        // 读取确认数

    // 写入
    public void write(String key, String value) throws Exception {
        // 写入所有节点
        List<CompletableFuture<Boolean>> futures = new ArrayList<>();
        for (Node node : nodes) {
            futures.add(node.put(key, value));
        }

        // 等待 W 个节点确认
        int count = 0;
        for (CompletableFuture<Boolean> future : futures) {
            if (future.get()) count++;
        }

        if (count < W) {
            throw new WriteException("写入未获得足够确认");
        }
    }

    // 读取
    public String read(String key) throws Exception {
        // 读取 R 个节点
        List<CompletableFuture<String>> futures = new ArrayList<>();
        for (Node node : nodes) {
            futures.add(node.get(key));
        }

        // 取最新版本
        String latest = null;
        long version = 0;
        for (CompletableFuture<String> future : futures) {
            Pair<String, Long> result = parse(future.get());
            if (result.getValue() > version) {
                latest = result.getKey();
                version = result.getValue();
            }
        }

        return latest;
    }
}
```

## 总结

强一致性有多种实现方案。

**单主复制**：

- 实现简单
- 延迟低
- 主库是瓶颈

**共识算法（Raft/Paxos）**：

- 高可用
- 强一致性
- 实现复杂

**Quorum 机制**：

- 可配置一致性级别
- 读写性能可调
- 需要多数派写入

**选择建议**：

- 小规模系统：单主复制
- 高可用系统：共识算法
- 读写分离系统：Quorum 机制
